  • LIKE

    • select * from mytable where name like "robot%";

  • union

    • select name from mytable union select name from mytable2;
    • select name from mytable union all select name from mytable2; //统计相同字段个数
    • union all显示重复的 union不显示

  • order by asc/desc

    • select * from mytable order by score;

  • group by

    • select name from mytable group by name; //查询字段要有group by内容

  • [inner] join … on … //inner可省略 组合多个表内容

    • select name, age, father, mony, familycount from mytable inner join mytable2 on mytable.id = mytable2.m_id;
    • 等价于
      • select name, age, father, mony,familycount from mytable,mytable2 where mytable.id = mytable2.m_id;

  • is null / is not null //判断字段是否为空

    • ifnull(列名，值) //替换值
      • select ifnull(name,’xxx’) from mytable;
